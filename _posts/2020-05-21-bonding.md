Bonding Configuration - Linux Space

Bonding Master 설정
cat /etc/sysconfig/network-scripts/ifcfg-bond0

DEVICE=bond0
BOOTRPOTO=autoip
MTU=1500
NAME=bond0
NM_CONTROLLED=no
ONBOOT=yes
USERCTL=no
TYPE=Bond
BONDING_MASTER=yes
BONDING_OPTS="mode=4 miimon=100"

Bonding Slave 설정
cat /etc/sysconfig/network-scripts/ifcfg-enp3s0
DEVICE=enp3s0
NAME=enp3s0
TYPE=Ethernet
ONBOOT=no
BOOTPROTO=none
MASTER=bond0
SLAVE=yes
NM_CONTROLLED=no
USERCTL=no
